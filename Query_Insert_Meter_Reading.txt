UPDATE dbo.R5OBJUSAGEDEFS SET
OUD_CREATED=GETDATE(),
OUD_LASTREADDATE=GETDATE(),
OUD_TOTALUSAGE=1000
WHERE OUD_OBJECT='MP3000' AND OUD_UOM='PCE' AND OUD_OBJECT_ORG='*'

INSERT INTO dbo.R5READINGS (REA_METER,REA_DATE,REA_READING,REA_UOM,
REA_CALCUOM,REA_OBJECT_ORG,REA_OBTYPE,REA_OBRTYPE,REA_PARENT,REA_OBJECT,
REA_DIFF,REA_SOURCESYSTEM,REA_SOURCECODE,REA_TRANSORGID,REA_TRANSCODE,REA_TRANSGROUP,
REA_CODE,REA_UPDATECOUNT,REA_CREATED,REA_UPDATED)
VALUES(NULL,GETDATE(),1000,'PCE',
'PCE','*','S','S',NULL,'MP3000',
NULL,NULL,NULL,NULL,NULL,NULL,
NULL,0,GETDATE(),GETDATE())

SELECT OUD_OBJECT, OUD_TOTALUSAGE AS 'OUD_USG',OUD_UPDATECOUNT+1 as 'OUD_UPC',
OUD_SINCEINSTALL AS 'OUD_SI', OUD_SINCELASTWORK AS 'OUD_SLW',
ISNULL((SELECT REA_DIFF FROM R5READINGS WHERE REA_SQLIDENTITY=(SELECT MAX(REA_SQLIDENTITY) 
FROM R5READINGS WHERE (REA_OBJECT ='MP3000') AND (REA_UOM ='HUR') AND 
(REA_OBJECT_ORG ='*'))),0) AS 'OUD_DIFF' FROM R5OBJUSAGEDEFS 
WHERE (OUD_OBJECT ='MP3000') AND (OUD_UOM ='HUR') AND (OUD_OBJECT_ORG ='*')
